  - name: "7 Naming Criteria"

      criterion_2_brevity:
        name: "Brevity"
        description: "Short enough to remember and use easily"
        test: "Is it four syllables or less?"
        scoring:
          strong: "1-2 syllables ideal (Nike, Apple)"
          acceptable: "3-4 syllables (Amazon, Microsoft)"
          weak: "5+ syllables (complicated to say)"
        weight: "High - longer names get shortened anyway"

      criterion_3_appropriateness:
        name: "Appropriateness"
        description: "Fits the brand without being too generic"
        test: "Does it suggest the category without describing it?"
        scoring:
          strong: "Suggestive but not literal (Safari for browser)"
          weak: "Too generic (Discount Furniture Warehouse)"
        weight: "High - balance between fit and uniqueness"

      criterion_4_easy_spelling_pronunciation:

        test: "Can people spell it after hearing it once?"
        scoring:
          strong: "Phonetic, intuitive spelling (Slack, Zoom)"
          weak: "Confusing spelling (Lyft vs Lift)"
        weight: "High - especially for word-of-mouth"

      criterion_5_likability:
        name: "Likability"
        description: "Creates positive emotional response"
        test: "Do people smile or feel good when they hear it?"
        scoring:
          strong: "Pleasant associations (Innocent, Smile)"
          weak: "Negative or neutral associations"
        weight: "Medium - contributes to brand warmth"

      criterion_6_extendability:
        name: "Extendability"
        description: "Has 'legs' for creative execution"
        test: "Can you build a visual world around it?"
        scoring:
          strong: "Rich metaphor potential (Amazon, Apple)"
          weak: "Limited creative possibilities"
        weight: "Medium - important for brand system"

      criterion_7_protectability:
        name: "Protectability"
        description: "Can be legally trademarked"
        test: "Is it unique enough to protect legally?"
        scoring:
          strong: "Coined words, unique combinations (Kodak)"
          weak: "Generic terms, common words"
        weight: "Critical - without this, name is vulnerable"
